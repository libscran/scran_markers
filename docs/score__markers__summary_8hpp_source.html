297void preallocate_minrank_queues(
298 const std::size_t ngroups,
299 MinrankTopQueues<Stat_, Index_>& queue,
300 const std::vector<SummaryBuffers<Stat_, Rank_> >& summaries,
301 const Index_ limit,
302 const bool keep_ties
303) {
304 topicks::TopQueueOptions<Stat_> qopt;
305 qopt.keep_ties = keep_ties;
306 qopt.check_nan = true;
307
308 sanisizer::resize(queue, ngroups);
309 for (I<decltype(ngroups)> g1 = 0; g1 < ngroups; ++g1) {
310 if (summaries[g1].min_rank == NULL) {
311 continue;
312 }
313 queue[g1].emplace();
314
315 auto& g_queue = *(queue[g1]);
316 sanisizer::reserve(g_queue, ngroups);
317 for (I<decltype(ngroups)> g2 = 0; g2 < ngroups; ++g2) {
318 g_queue.emplace_back(limit, true, qopt);
319 }
320 }

349template<typename Stat_, typename Index_, typename Rank_>
350void report_minrank_from_queues(
351 const Index_ ngenes,
352 const std::size_t ngroups,
353 std::vector<std::optional<MinrankTopQueues<Stat_, Index_> > >& all_queues,
354 const std::vector<SummaryBuffers<Stat_, Rank_> >& summaries,
355 const int num_threads,
356 const bool keep_ties
357) {
358 if (all_queues.empty()) {
359 // If no queues were populated with ranks, this means that the matrix had no rows at all.
360 // If that's the case, there's no point iterating through the groups.
361 // We don't need to fill the min_rank array because ngenes == 0.
362 // Thus, we can just return immediately.
363 return;
364 }
365
366 tatami::parallelize([&](const int, const std::size_t start, const std::size_t length) -> void {
367 std::vector<Index_> tie_buffer;
368
369 for (I<decltype(ngroups)> gr = start, grend = start + length; gr < grend; ++gr) {
370 const auto mr_out = summaries[gr].min_rank;
371 if (mr_out == NULL) {
372 continue;
373 }
374
375 // Using the maximum possible rank (i.e., 'ngenes') as the default.
376 const auto maxrank_placeholder = sanisizer::cast<Rank_>(ngenes);
377 std::fill_n(mr_out, ngenes, maxrank_placeholder);
378
379 auto& first_mr_queue = *(all_queues.front());
380 for (I<decltype(ngroups)> gr2 = 0; gr2 < ngroups; ++gr2) {
381 if (gr == gr2) {
382 continue;
383 }
384 auto current_out = std::move((*(first_mr_queue[gr]))[gr2]); // moving it to minimize false sharing.
385
386 const auto num_queues = all_queues.size();
387 for (I<decltype(num_queues)> q = 1; q < num_queues; ++q) {
388 auto& current_mr_queue = *(all_queues[q]);
389 auto current_in = std::move((*(current_mr_queue[gr]))[gr2]); // moving it to minimize false sharing.
390 while (!current_in.empty()) {
391 current_out.push(current_in.top());
392 current_in.pop();
393 }
394 }
395
396 // Cast to Rank_ is safe as current_out.size() <= ngenes,
397 // and we already checked that ngenes can fit into Rank_ in report_minrank_from_current_outs().
398 if (!keep_ties) {
399 while (!current_out.empty()) {
400 auto& mr = mr_out[current_out.top().second];
401 mr = std::min(mr, static_cast<Rank_>(current_out.size()));
402 current_out.pop();
403 }
404 } else {
405 while (!current_out.empty()) {
406 tie_buffer.clear();
407 const auto curtop = current_out.top();
408 current_out.pop();
409
410 while (!current_out.empty() && current_out.top().first == curtop.first) {
411 tie_buffer.push_back(current_out.top().second);
412 current_out.pop();
413 }
414
415 // Increment is safe as we already reduced the size at least once.
416 const Rank_ tied_rank = current_out.size() + 1;
417
418 mr_out[curtop.second] = std::min(mr_out[curtop.second], tied_rank);
419 for (const auto t : tie_buffer) {
420 mr_out[t] = std::min(mr_out[t], tied_rank);
421 }
422 }
423 }
424 }
425 }
426 }, ngroups, num_threads);
427}
