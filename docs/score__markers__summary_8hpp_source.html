288template<typename Stat_, typename Index_, typename Rank_>
289void preallocate_minrank_queues(
290 const std::size_t ngroups,
291 std::vector<MinrankTopQueues<Stat_, Index_> >& all_queues,
292 const std::vector<SummaryBuffers<Stat_, Rank_> >& summaries,
293 const Index_ limit,
294 const bool keep_ties,
295 const int num_threads
296) {
297 topicks::TopQueueOptions<Stat_> qopt;
298 qopt.keep_ties = keep_ties;
299 qopt.check_nan = true;
300
301 sanisizer::resize(all_queues, num_threads);
302 for (int t = 0; t < num_threads; ++t) {
303 sanisizer::resize(all_queues[t], ngroups);
304
305 for (decltype(I(ngroups)) g1 = 0; g1 < ngroups; ++g1) {
306 if (summaries[g1].min_rank == NULL) {
307 continue;
308 }
309
310 for (decltype(I(ngroups)) g2 = 0; g2 < ngroups; ++g2) {
311 all_queues[t][g1].emplace_back(limit, true, qopt);
312 }
313 }
314 }

317template<typename Stat_, typename Index_, typename Rank_>
318void compute_summary_stats_per_gene(
319 const Index_ gene,
320 const std::size_t ngroups,
321 const Stat_* const pairwise_buffer_ptr,
322 std::vector<Stat_>& summary_buffer,
323 MinrankTopQueues<Stat_, Index_>& minrank_queues,
324 const std::vector<SummaryBuffers<Stat_, Rank_> >& summaries
325) {

327 auto& cursummary = summaries[gr];
328 const auto in_offset = sanisizer::product_unsafe<std::size_t>(ngroups, gr);

342template<typename Stat_, typename Index_, typename Rank_>
343void report_minrank_from_queues(
344 const Index_ ngenes,
345 const std::size_t ngroups,
346 std::vector<MinrankTopQueues<Stat_, Index_> >& all_queues,
347 const std::vector<SummaryBuffers<Stat_, Rank_> >& summaries,
348 const int num_threads,
349 const bool keep_ties
350) {
351 tatami::parallelize([&](const int, const std::size_t start, const std::size_t length) -> void {
352 std::vector<Index_> tie_buffer;
353
354 for (decltype(I(ngroups)) gr = start, grend = start + length; gr < grend; ++gr) {
355 const auto mr_out = summaries[gr].min_rank;
356 if (mr_out == NULL) {
357 continue;
358 }
359
360 // Using the maximum possible rank (i.e., 'ngenes') as the default.
361 const auto maxrank_placeholder = sanisizer::cast<Rank_>(ngenes);
362 std::fill_n(mr_out, ngenes, maxrank_placeholder);
363
364 for (decltype(I(ngroups)) gr2 = 0; gr2 < ngroups; ++gr2) {
365 if (gr == gr2) {
366 continue;
367 }
368 auto& current_out = all_queues.front()[gr][gr2];
369
370 const auto num_queues = all_queues.size();
371 for (decltype(I(num_queues)) q = 1; q < num_queues; ++q) {
372 auto& current_in = all_queues[q][gr][gr2];
373 while (!current_in.empty()) {
374 current_out.push(current_in.top());
375 current_in.pop();
376 }
377 }
378
379 // Cast to Rank_ is safe as current_out.size() <= ngenes,
380 // and we already checked that ngenes can fit into Rank_ in report_minrank_from_current_outs().
381 if (!keep_ties) {
382 while (!current_out.empty()) {
383 auto& mr = mr_out[current_out.top().second];
384 mr = std::min(mr, static_cast<Rank_>(current_out.size()));
385 current_out.pop();
386 }
387 } else {
388 while (!current_out.empty()) {
389 tie_buffer.clear();
390 const auto curtop = current_out.top();
391 current_out.pop();
392
393 while (!current_out.empty() && current_out.top().first == curtop.first) {
394 tie_buffer.push_back(current_out.top().second);
395 current_out.pop();
396 }
397
398 // Increment is safe as we already reduced the size at least once.
399 const Rank_ tied_rank = current_out.size() + 1;
400
401 mr_out[curtop.second] = std::min(mr_out[curtop.second], tied_rank);
402 for (const auto t : tie_buffer) {
403 mr_out[t] = std::min(mr_out[t], tied_rank);
404 }
405 }
406 }
407 }
408 }
409 }, ngroups, num_threads);
410}
