251 const std::optional<Stat_>& bound

269template<typename Stat_, typename Index_>
270void add_best_top_queues(
271 PairwiseTopQueues<Stat_, Index_>& pqueues,
272 const Index_ gene,
273 std::size_t ngroups,
274 const std::vector<Stat_>& effects
275) {
276 for (I<decltype(ngroups)> g1 = 0; g1 < ngroups; ++g1) {

286template<typename Stat_, typename Index_>
287void report_best_top_queues(
288 std::vector<std::optional<PairwiseTopQueues<Stat_, Index_> > >& pqueues,
289 std::size_t ngroups,
290 std::vector<std::vector<std::vector<std::pair<Index_, Stat_> > > >& output
291) {
292 // We know it fits into an 'int' as this is what we got originally.
293 const int num_available = pqueues.size();
294
295 // If it's empty, we just create empty vectors and move on.
296 if (num_available == 0) {
297 sanisizer::resize(output, ngroups);
298 for (I<decltype(ngroups)> g1 = 0; g1 < ngroups; ++g1) {
299 sanisizer::resize(output[g1], ngroups);
300 }
301 return;
302 }
303
304 // Consolidating all of the thread-specific queues into a single queue.
305 auto& true_pqueue = *(pqueues.front());
306 for (int t = 1; t < num_available; ++t) {
307 auto& current_pqueue = *(pqueues[t]);
308 for (I<decltype(ngroups)> g1 = 0; g1 < ngroups; ++g1) {
309 for (I<decltype(ngroups)> g2 = 0; g2 < ngroups; ++g2) {
310 auto& current_in = current_pqueue[g1][g2];
311 auto& current_out = true_pqueue[g1][g2];
312 while (!current_in.empty()) {
313 current_out.push(current_in.top());
314 current_in.pop();
315 }
316 }
317 }
318 }
319
320 // Now spilling them out into a single vector.
321 sanisizer::resize(output, ngroups);
322 for (I<decltype(ngroups)> g1 = 0; g1 < ngroups; ++g1) {
323 sanisizer::resize(output[g1], ngroups);
324 for (I<decltype(ngroups)> g2 = 0; g2 < ngroups; ++g2) {
325 if (g1 == g2) {
326 continue;
327 }
328 auto& current_in = true_pqueue[g1][g2];
329 auto& current_out = output[g1][g2];
330 while (!current_in.empty()) {
331 const auto& best = current_in.top();
332 current_out.emplace_back(best.second, best.first);
333 current_in.pop();
334 }
335 std::reverse(current_out.begin(), current_out.end()); // earliest element should have the strongest effect sizes.
336 }
337 }
